{
    return t1.a<t2.a;
}
bool cmp2(abc t1,abc t2)
{
    return t1.b<t2.b;
}
bool cmp3(abc t1,abc t2)
{
    return t1.c<t2.c;

{
    int N;

    {
        for(int t=1;t<=N;t++)
        {
            int i,j,k;

            item[t].a=i-j;
            item[t].b=j-k;
            item[t].c=k-i;
        }
        long long ans=0;
        sort(item+1,item+N+1,cmp1);
        for(long long i=1;i<=N;i++)
        {
            ans+=(i-1)*(item[i].a);
            ans-=(N-i)*(item[i].a);
        }

把a、b、c想成是数轴上的三个点就很容易得出上述公式了。

然而求出所有的|a-b|+|b-c|+|a-c|，再求和也不是简单的问题，所以进而继续化简得：

max {Ia ? Ib, Ja ? Jb, Ka ? Kb} ? min {Ia ? Ib, Ja ? Jb, Ka ? Kb}

=(|(Ia ? Ib)―( Ja C Jb)|+|(Ja C Jb)―(Ka ? Kb)|+|( Ka ? Kb)-( Ia ? Ib)|)/2

= (|(Ia C Ja)―(Ib―Jb)|+|(Ja ? Ka)―(Jb ? Kb)|+| (Ka C Ia)―( Ka ? Ib)|)/2
