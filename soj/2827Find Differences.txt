#include<stdio.h>

int main(void)
{
    int result;
    int now;
    int n;
    while(EOF!=scanf("%d",&n)&&n)    // 水的一比。。。。
    {
        int i;
        result=0;
        for(i=1;i<=(2*n+1);i++)
        {
            scanf("%d",&now);
            result=(result^now);
        }
        printf("%d\n",result);
    }
    return 0;
}
