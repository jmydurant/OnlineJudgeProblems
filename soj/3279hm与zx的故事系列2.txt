void gongzuo(int point)
{
    int i=point;
    int j=point;
    while(1)
    {
        if(0==number[j].right)
        {
            result[i]=0;
            return ;
        }
        else
        {
            j=number[j].right;
            if(number[j].num>number[i].num)
            {
                result[i]=number[j].num;
                return;
            }
            else
            {
                number[i].right=number[j].right;

            }
        }

    }
}
