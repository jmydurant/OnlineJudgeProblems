using namespace std;
int visit[10010];
int digit[10010];
int main(void)
{
    int N;
    while(EOF!=scanf("%d",&N))
    {
        for(int i=1;i<=N;i++)
            scanf("%d",digit+i);
        int ans=0;
        memset(visit,-1,sizeof(visit));
        visit[0]=0;
        int left,right;
        for(int i=1;i<=N;i++)
        {
            ans+=digit[i];
            ans%=N;
            if(-1!=visit[ans])
            {
                left=visit[ans];
                right=i;
                break;
            }
            else
            {
                visit[ans]=i;
            }
        }
        printf("%d\n",right-left);
        for(int i=left+1;i<=right;i++)
        {
            printf("%d\n",digit[i]);
        }
    }
    return 0;
}
