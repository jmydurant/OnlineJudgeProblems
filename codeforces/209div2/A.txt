int a[100][100];
int row,col;
bool judge(int i,int j)
{
    if(i == 1 || i == row || j == 1 || j == col)
        return true;
    return false;
}
int main(void)
{
    while(2 == scanf("%d%d",&row,&col))
    {
        bool ok = false;
        for(int i = 1;i<=row;i++)
        {
            for(int j = 1;j<=col;j++)
            {
                scanf("%d",&a[i][j]);
                if(a[i][j] && judge(i,j))
                    ok = true;
            }
        }
        if(ok)
            printf("2\n");
        else
            printf("4\n");
    }

}
