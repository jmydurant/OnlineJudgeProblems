using namespace std;
typedef long long LL;

LL dp[26][maxn];
char a[maxn],b[maxn];
LL ta,tb;

LL gcd(LL x,LL y)
{
    LL temp;
    while(y)
    {
        temp = x%y;
        x = y;
        y = temp;
    }
    return x;
}

LL lcm(LL x,LL y)
{
    return x/gcd(x,y)*y;
}

int main(void)
{
    while(2 == scanf("%I64d%I64d",&ta,&tb))
    {
        scanf("%s",a);
        scanf("%s",b);
        LL alen = strlen(a);
        LL blen = strlen(b);
        LL ggcd = gcd(alen,blen);
        LL llcm = lcm(alen,blen);

        REP(i,blen)

            ans -= dp[a[i]-'a'][i%ggcd];
        }

    }
